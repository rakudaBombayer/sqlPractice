ほかのテーブルからデータをコピーする
CREATE TABLE ShohinCopy
(shohin_id CHAR(4) NOT NULL,
 shohin_mei VARCHAR(100) NOT NULL,

shohin_bunrui VARCHAR(32) NOT NULL,

hanbai_tanka INTEGER,

shiire_tanka INTEGER,
torokubi DATE ,
 PRIMARY KEY (shohin_id));


INSERT INTO ShohinCopy (
  shohin_id, shohin_mei, shohin_bunrui,
  hanbai_tanka, shiire_tanka, torokubi
) SELECT shohin_id, shohin_mei, shohin_bunrui,
  hanbai_tanka, shiire_tanka, torokubi FROM Shohin;

→INSERT INTO ShohinCopy SELECT * FROM Shohin;(全件同じならこれでも可)
→INSERT INTO ShohinCopy SELECT * FROM Shohin WHERE shouhin_bunrui ='キッチン用品';
(条件付きでコピーできる)


CREATE TABLE ShohinBunrui
(shohin_bunrui VARCHAR(32) NOT NULL,

sum_hanbai_tanka INTEGER,

sum_shiire_tanka INTEGER,

 PRIMARY KEY (shohin_bunrui));



INSERT INTO ShohinBunrui (
  shohin_bunrui, 
  sum_hanbai_tanka, sum_shiire_tanka
) SELECT shohin_bunrui, SUM(hanbai_tanka), SUM(shiire_tanka) FROM Shohin
GROUP BY shohin_bunrui;
